LinkedList


struct Node
{
int val;
Node* next;  //Valid because compiler knows that Node exists
//Node next; //Not allowed
};

void printList(Node* head){
Node* curr=head; //Cannot lose the head pointer because creates a memory leak, should not modify linked list
while(curr!=nullptr){
cout<<curr->value<<endl;
curr=curr->next;
}
}

Node* search(Node* head,int val){
Node* p;
for(p=head; p!=nullptr && p->value!= val; p=p->next){
return p;
}
return nullptr;
}

//Insert value after target value, returns true if successful
bool insert(Node* head,int target, int val){ 
Node* p;
for(p=head; p!=nullptr && p->value!= target; p=p->next){
;
}
if(p!=nullptr){
Node* newNode=new Node;
newNode->value=val;
newNode->next=p->next;
p->next=newNode;

return true;
}
return false;
}
