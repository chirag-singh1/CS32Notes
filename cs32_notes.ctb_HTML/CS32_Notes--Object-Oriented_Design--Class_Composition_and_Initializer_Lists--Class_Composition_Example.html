Class Composition Example


class Stomach{
public: 
Stomach(){myGas=0;}
void eat(){myGas++;}
~Stomach(){cout<<"a";}
...
};
class Brain{
public:
Brain(){myIq=100;}
void think(){myIQ+=10;}
~Brain(){cout<<"b";}
};
class HungryNerd{
public:
~HungryNerd(){
cout<<"c";
//Implicitly adds calls to Brain and Stomach constructors
}
HungryNerd(){
//Will first call the default constructros of Brain and Stomach implicitly
myBelly.eat();
myBrain.think();
}
private:
Stomach myBelly; //Constructed first
Brain myBrain;
}

class Stomach{
public: 
Stomach(int startGas){myGas=startGas;} //Default constructor now creates undefined behavior
void eat(){myGas++;}
~Stomach(){cout<<"a";}
...
};
class Brain{
public:
Brain(int startIq){myIq=startIq;}
void think(){myIQ+=10;}
~Brain(){cout<<"b";}
};
class HungryNerd{
public:
~HungryNerd(){
cout<<"c";
//Implicitly adds calls to Brain and Stomach constructors
}
HungryNerd():myBelly(10),myBrain(150),myAge(19){ //This is an initializer list
//Problem: autogenerated constructor for Stomach will call on default
//Solution: explicitly call the constructor for Stomach (with parameter)
myBelly.eat();
myBrain.think();
}
private:
Stomach myBelly; //Constructed first
Brain myBrain;
int myAge;
}
