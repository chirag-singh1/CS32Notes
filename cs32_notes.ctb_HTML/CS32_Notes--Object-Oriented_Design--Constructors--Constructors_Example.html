Constructors Example


class Gassy
{
public:
Gassy(int age, bool ateBeans){
m_Age=age;
m_ateBeans=ateBeans;
}
Gassy() //overloaded constructor, this constructor will be generated if it does not exist
{
m_Age=0;
m_ateBeans=false;
}

private:
int m_age;
boolean m_ateBeans;
}
int main()
{
Gassy carey; //Notice that parentheses are not used or they would declare a function
Gassy arr[52]; //Calls default constructor 52 times
Gassy carey(45,false);
Gassy *ptr=new Gassy(1,false) //This is dynamic memory allocation (new keyword)
Gassy *ptr; //This is a null pointer until the constructor is called
}
