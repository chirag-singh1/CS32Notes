Separate Compilation Example


Point.h
=============
#ifndef POINT //Include Guard for Point class
#define POINT

class Point{
...
};

#endif

Circle.h
=============
#ifndef CIRCLE //All header files should have include guards
#define CIRCLE

#include "Point.h" //Circle should have to include Point.h to compile
class Circle{
public:
...
private:
Point m_center;
double m_radius;
}

#endif

myapp.cpp
==============
#include "Circle.h"
#include "Point.h" //Point.h must also be included, but this creates 2 definitions
int main(){
Circle c(-2,5,10);
Point p(2,4);
}

Course.h
========
#ifndef COURSE
#define COURSE

{
...
